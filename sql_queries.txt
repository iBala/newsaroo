Here are the schemas available in supabase for this newsaroo app

Table 1 - newsaroo_users - This stores the mobile no and topics of interest of the user. 

CREATE TABLE public.newsaroo_users (
    id bigint GENERATED BY DEFAULT AS IDENTITY NOT NULL,
    created_at timestamp with time zone NOT NULL DEFAULT now(),
    name text NOT NULL,
    mobile_no bigint NOT NULL,
    topics_of_interest jsonb NULL,
    CONSTRAINT newsaroo_users_pkey PRIMARY KEY (id)
);
